This is an example of method overloading in PHP5. We put here for
reference only.

function __call($metodo, $atributos){
  switch($metodo){
    case "funcion":
      $funcion = "$metodo_$tipo";
      $this->$funcion($atributos[0]);
      break;
    default:
      break;
  }
}
